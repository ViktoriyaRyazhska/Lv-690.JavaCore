    public void ArithmeticTest() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        System.out.println("Input variable a:_");
        String input1 = br.readLine();
        System.out.println("Input variable b:_");
        String input2 = br.readLine();

        double a = Double.parseDouble(input1);
        double b = Double.parseDouble(input2);

        double sum = a+b;
        double dif = a-b;

        double x = a/b;
        double y = a*b;

        System.out.println("Your result a+b=" +sum);
        System.out.println("Your result a-b=" +dif);

    public void GeometryTest() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        System.out.println("Input radius of the circle:_");
        String radius = br.readLine();
        double r = Double.parseDouble(radius);
        double perimeter = 2*r*Math.PI;
        double area = Math.PI*Math.pow(r,2);
        System.out.println("Perimeter of the circle = "+perimeter);
        System.out.println("Area of the circle = "+area);
    }
