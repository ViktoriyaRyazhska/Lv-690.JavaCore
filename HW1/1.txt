    public void OnlyPrint(String txt){
        System.out.println(txt);
    }
    public String PrintWithInput(String txt)throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        System.out.println(txt);
        return br.readLine();
    }
    public void ArithmeticTest() throws IOException {

        String input1 = PrintWithInput("Input variable :_");
        String input2 = PrintWithInput("Input variable b:_");

        double a = Double.parseDouble(input1);
        double b = Double.parseDouble(input2);

        double sum = a+b;
        double dif = a-b;

        double x = a/b;
        double y = a*b;

        OnlyPrint("Your result a+b=" +sum);
        OnlyPrint("Your result a-b=" +dif);
        OnlyPrint("Your result a/b=" +x);

    }
    public void GeometryTest()throws IOException {

        String radius = PrintWithInput("Input radius of the circle:_");

        double r = Double.parseDouble(radius);
        double perimeter = 2*r*Math.PI;
        double area = Math.PI*Math.pow(r,2);

        OnlyPrint("Perimeter of the circle = "+perimeter);
        OnlyPrint("Area of the circle = "+area);
    }
