package JavaCore;

public class Employee {

    private String name;
    private int rate;
    private double hours;

    public static double totalSum;

    protected Employee(String name, int rate, double hours) {
        this.name = name;
        this.rate = rate;
        this.hours = hours;

        totalSum+=this.getSalary();

    }

    protected Employee(String name, int rate) {
        this.name = name;
        this.rate = rate;
    }

    protected Employee() {  }

    protected String getName() {
        return name;
    }

    protected void setName(String name) {
        this.name = name;
    }

    protected int getRate() {
        return rate;
    }

    protected void setRate(int rate) {
        totalSum-=this.getSalary();
        this.rate = rate;
        totalSum+=this.getSalary();

    }

    protected double getHours() {
        return hours;
    }

    protected void setHours(double hours) {
        totalSum-=this.getSalary();
        this.hours = hours;
        totalSum+=this.getSalary();
    }

    @Override
    public String toString() {
        return "Employee{" +
                "name='" + name + '\'' +
                ", rate=" + rate +
                ", hours=" + hours +
                '}';
    }

    protected double getSalary(){
        return this.hours*this.rate;
    }
    protected double changeRate(int r){
        totalSum-=this.getSalary();
        this.rate=r;
        totalSum+=this.getSalary();

       double salary=this.getSalary();
       return salary;
    }

    protected double getBonuses(){
        double salary=this.getSalary();
        double bonus = salary*0.1;
        return bonus;
    }
}
